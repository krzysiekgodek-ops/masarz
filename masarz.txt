            {/* ---- UŻYTKOWNICY ---- */}
            {adminSubTab === 'users' && (
              <div className="space-y-3">
                {allUsers.map(u => {
                  const uTrialActive = u.createdAt ? Math.floor((new Date() - new Date(u.createdAt)) / 86400000) <= 21 : false;
                  const uRecipes = allUserRecipes.filter(r => r.ownerId === u.id).length;
                  const uPlanData = (plans || DEFAULT_PLANS).food[u.plan || 'free'];

                      <div className="flex items-start justify-between gap-4 flex-wrap">

                          <div className="w-10 h-10 rounded-2xl bg-blue-100 flex items-center justify-center text-blue-600 font-black text-sm shrink-0">

                          </div>
                          <div className="min-w-0">
                            <p className="font-black text-sm text-slate-900 truncate">{u.email}</p>

                              Dołączył: {u.createdAt ? new Date(u.createdAt).toLocaleDateString('pl-PL') : '—'}

                          {uTrialActive && <span className="bg-blue-100 text-blue-600 text-[9px] font-black px-2.5 py-1 rounded-full uppercase">Trial</span>}

                          <select value={u.plan || 'free'} onChange={(e) => updatePlayerPlan(u.id, e.target.value)} className="text-xs font-black uppercase bg-transparent outline-none cursor-pointer text-slate-900 w-full text-center">
                            <option value="free">Free</option>

                        </div>

                      {u.email !== SUPER_ROOT && (

                          <button onClick={() => toggleAdmin(u.id, u.isAdmin)} className={`flex-1 py-2 rounded-xl text-[10px] font-black uppercase transition-all ${u.isAdmin ? 'bg-amber-100 text-amber-700 hover:bg-amber-200' : 'bg-slate-100 text-slate-500 hover:bg-slate-200'}`}>
                            {u.isAdmin ? 'Odbierz Admina' : 'Nadaj Admina'}
                          </button>
                          <button onClick={() => deleteUserAccount(u.id, u.email)} className="px-4 py-2 bg-red-50 text-red-500 rounded-xl hover:bg-red-600 hover:text-white transition-all" title="Usuń konto">
                            <Trash2 size={14}/>
                          </button>

                      )}
                      {u.email === SUPER_ROOT && (
                        <div className="mt-3 text-center">
                          <span className="text-[10px] font-black text-blue-600 uppercase tracking-widest">Właściciel systemu</span>

                    </div>
                  );
                })}
              </div>
            )}

            {/* ---- BANERY ---- */}
            {adminSubTab === 'banners' && (
              <div className="space-y-4">
                <div className="flex items-center justify-between">
                  <div>
                    <h2 className="text-xl font-black uppercase italic tracking-tighter text-slate-900">Banery reklamowe</h2>

                      <div className="h-36 bg-slate-100 overflow-hidden">

                            onChange={(e) => updateDoc(doc(db, 'ads', ad.id), { content: e.target.value })}

                          <button

                          </button>

                            Usuń

                <div className="flex items-center justify-between flex-wrap gap-3">

                    <h2 className="text-xl font-black uppercase italic tracking-tighter text-slate-900">Receptury użytkowników</h2>
                    <p className="text-xs text-slate-500 mt-1">{allUserRecipes.filter(r => r.ownerId !== 'ADMIN').length} receptur · {stats.blockedRecipes} zablokowanych · {stats.verifiedRecipes} zweryfikowanych</p>

                            {r.imageUrl ? <img src={r.imageUrl} className="w-full h-full object-cover" alt=""/> : <ChefHat size={18} className="text-slate-300"/>}
                          </div>
                          <div className="flex-1 min-w-0">
                            <div className="flex items-center gap-2 flex-wrap">
                              <h4 className="font-black text-slate-900 text-sm">{r.name}</h4>

                              {r.verified && <span className="bg-green-100 text-green-700 text-[9px] font-black px-2 py-0.5 rounded-full uppercase">Zweryfikowana</span>}
                            </div>
                            <p className="text-[10px] text-slate-400 font-bold uppercase mt-0.5">{r.category}</p>
                            <p className="text-[10px] text-blue-500 font-bold mt-1 truncate">{owner?.email || r.ownerId}</p>
                          </div>
                          <div className="flex flex-col gap-1.5 shrink-0">
                            <button
                              onClick={() => setSelectedModRecipe(isExpanded ? null : r)}

                              <Eye size={11}/> {isExpanded ? 'Zwiń' : 'Podgląd'}
                            </button>
                            <button
                              onClick={() => toggleBlockRecipe(r.id, r.blocked)}
                              className={`flex items-center gap-1.5 px-3 py-1.5 rounded-xl text-[10px] font-black uppercase transition-all ${r.blocked ? 'bg-orange-100 text-orange-700 hover:bg-orange-200' : 'bg-red-50 text-red-600 hover:bg-red-100'}`}
                            >
                              <Ban size={11}/> {r.blocked ? 'Odblokuj' : 'Zablokuj'}
                            </button>
                            <button
                              onClick={() => verifyRecipe(r.id, r.verified)}
                              className={`flex items-center gap-1.5 px-3 py-1.5 rounded-xl text-[10px] font-black uppercase transition-all ${r.verified ? 'bg-green-200 text-green-800 hover:bg-green-300' : 'bg-green-50 text-green-700 hover:bg-green-100'}`}
                            >
                              <ShieldCheck size={11}/> {r.verified ? 'Cofnij' : 'Weryfikuj'}
                            </button>
                            <button
                              onClick={() => deleteRecipeAdmin(r.id)}
                              className="flex items-center gap-1.5 px-3 py-1.5 bg-red-600 text-white rounded-xl text-[10px] font-black uppercase hover:bg-red-700 transition-all"

                              <Trash2 size={11}/> Usuń

                                <p className="text-[9px] font-black uppercase text-slate-400 tracking-wider mb-2">Surowce mięsne</p>
                                <div className="space-y-1.5">
                                  {(r.meats || []).map((m, i) => (
                                    <div key={i} className="flex justify-between bg-white p-2.5 rounded-xl border text-xs">
                                      <span className="font-bold text-slate-800">{m.name}</span>
                                      <span className="font-black text-blue-600">{m.percentage?.toFixed(1)}%</span>

                                </div>
                              </div>
                              <div>
                                <p className="text-[9px] font-black uppercase text-slate-400 tracking-wider mb-2">Przyprawy / Dodatki</p>
                                <div className="space-y-1.5">
                                  {(r.spices || []).map((s, i) => (
                                    <div key={i} className="flex justify-between bg-slate-900 text-white p-2.5 rounded-xl text-xs">
                                      <span className="font-bold text-blue-300">{s.name}</span>
                                      <span className="font-black">{s.perKg} {s.unit}/kg</span>
                                    </div>
                                  ))}
                                </div>
                              </div>
                            </div>
                            {r.tech && (
                              <div>
                                <p className="text-[9px] font-black uppercase text-slate-400 tracking-wider mb-2">Opis technologiczny</p>
                                <div className="bg-white p-4 rounded-2xl border text-xs text-slate-700 leading-relaxed font-medium italic" dangerouslySetInnerHTML={{ __html: formatTechText(r.tech) }} />
                              </div>
                            )}
                          </div>
                        )}
                      </div>
                    );
                  })}

                {allUserRecipes.filter(r => r.ownerId !== 'ADMIN').filter(r => !modFilterUser || r.ownerId === modFilterUser).length === 0 && (
                  <div className="text-center py-16 bg-white rounded-[2rem] border">
                    <Shield size={40} className="mx-auto mb-4 text-slate-200"/>
                    <p className="font-bold text-sm text-slate-400 uppercase">Brak receptur do moderacji</p>
                  </div>
                )}
              </div>
            )}

            {/* ---- CENNIK & KATEGORIE ---- */}
            {adminSubTab === 'pricing' && plans && (
              <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
                <div className="bg-white p-6 rounded-[2rem] shadow-sm border text-left">
                  <h3 className="text-base font-black uppercase mb-5 text-blue-600 flex items-center gap-2"><Tag size={16}/> Cennik planów</h3>
                  {['mini', 'midi', 'max'].map(k => (
                    <div key={k} className="mb-4 bg-slate-50 p-4 rounded-2xl border">
                      <p className="text-[10px] font-black uppercase text-slate-400 mb-2">{k}</p>
                      <div className="flex gap-3">
                        <input className="bg-white border p-3 rounded-xl flex-1 font-bold outline-none text-sm" value={plans.food[k].price} onChange={(e) => updatePrice('food', k, {...plans.food[k], price: e.target.value})} />
                        <input className="bg-white border p-3 rounded-xl w-20 text-center font-black outline-none" type="number" value={plans.food[k].limit} onChange={(e) => updatePrice('food', k, {...plans.food[k], limit: Number(e.target.value)})} />
                      </div>
                    </div>
                  ))}
                </div>
                <div className="bg-[#0F172A] p-6 rounded-[2rem] text-white shadow-xl">
                  <h3 className="text-base font-black uppercase mb-5 text-blue-400 flex items-center gap-2"><Tag size={16}/> Kategorie Globalne</h3>
                  <div className="space-y-2 max-h-64 overflow-y-auto mb-4 pr-1">
                    {categories.map(c => (
                      <div key={c} className="flex justify-between items-center bg-white/5 p-3 rounded-xl border border-white/5 group">
                        <span className="text-xs font-bold uppercase tracking-widest">{c}</span>
                        <button onClick={async () => {
                          const snap = await getDocs(query(collection(db, 'categories'), where('name', '==', c)));
                          snap.forEach(async (d) => await deleteDoc(doc(db, 'categories', d.id)));
                        }} className="text-red-400 opacity-40 group-hover:opacity-100 p-1 transition-opacity"><Trash2 size={13}/></button>
                      </div>
                    ))}
                  </div>
                  <div className="flex gap-2">
                    <input placeholder="Nowa kategoria..." className="bg-white/10 rounded-xl p-3 text-xs w-full text-white outline-none focus:ring-2 focus:ring-blue-500" value={newCatName} onChange={e => setNewCatName(e.target.value)} onKeyDown={e => { if(e.key === 'Enter' && newCatName) { addDoc(collection(db, 'categories'), { name: newCatName }); setNewCatName(''); }}}/>
                    <button onClick={async () => { if(!newCatName) return; await addDoc(collection(db, 'categories'), { name: newCatName }); setNewCatName(''); }} className="bg-blue-600 p-3 rounded-xl hover:bg-blue-700 transition-all shrink-0"><Plus size={16}/></button>
                  </div>
                </div>

          </main>

        {/* MODAL: NOWY BANER */}
        {showBannerModal && (
          <div className="no-print fixed inset-0 bg-slate-900/90 z-50 flex items-center justify-center p-4 backdrop-blur-xl animate-in fade-in duration-200">
            <div className="bg-white rounded-[3rem] w-full max-w-lg p-8 relative shadow-2xl">
              <button onClick={() => setShowBannerModal(false)} className="absolute top-6 right-6 p-2 bg-slate-100 rounded-full hover:bg-slate-200 transition-colors"><X size={18}/></button>

                <div>
                  <p className="text-[10px] font-black text-slate-400 uppercase ml-2 mb-1">Treść baneru *</p>
                  <textarea
                    className="w-full p-4 border-2 rounded-2xl font-bold text-sm outline-none focus:border-blue-500 resize-none"
                    rows={3}
                    placeholder="Tekst wyświetlany na banerze..."
                    value={newBannerForm.content}
                    onChange={e => setNewBannerForm(p => ({...p, content: e.target.value}))}

                <div className="bg-slate-50 rounded-2xl border-2 border-dashed p-4 flex items-center justify-between gap-4">
                  {newBannerForm.imageUrl ? (
                    <div className="relative w-full">
                      <img src={newBannerForm.imageUrl} alt="" className="w-full h-24 object-cover rounded-xl"/>
                      <button onClick={() => setNewBannerForm(p => ({...p, imageUrl: ''}))} className="absolute top-1 right-1 bg-red-600 text-white p-1 rounded-full"><X size={10}/></button>

                    <label className="flex items-center gap-3 cursor-pointer w-full">
                      <div className="w-10 h-10 bg-blue-100 rounded-xl flex items-center justify-center shrink-0">
                        <ImageIcon size={16} className="text-blue-600"/>
                      </div>
                      <div>
                        <p className="text-[10px] font-black uppercase text-slate-600">{isBannerUploading ? 'Wgrywanie...' : 'Dodaj zdjęcie baneru'}</p>

                    </label>

                <div className="flex items-center gap-2">
                  <Link2 size={14} className="text-slate-400 shrink-0"/>
                  <input
                    type="url"
                    placeholder="Link do strony reklamodawcy (opcjonalnie)"
                    className="flex-1 p-3 border-2 rounded-xl text-sm font-medium outline-none focus:border-blue-500"
                    value={newBannerForm.linkUrl}
                    onChange={e => setNewBannerForm(p => ({...p, linkUrl: e.target.value}))}
                  />
                </div>

                <div className="flex items-center gap-2">
                  <FileIcon size={14} className="text-slate-400 shrink-0"/>
                  <input
                    type="url"
                    placeholder="Link do pliku PDF (opcjonalnie)"
                    className="flex-1 p-3 border-2 rounded-xl text-sm font-medium outline-none focus:border-blue-500"
                    value={newBannerForm.pdfUrl}
                    onChange={e => setNewBannerForm(p => ({...p, pdfUrl: e.target.value}))}
                  />
                </div>

                <div className="flex items-center justify-between p-4 bg-slate-50 rounded-2xl border">
                  <span className="text-sm font-bold text-slate-700">Aktywny od razu</span>
                  <button
                    onClick={() => setNewBannerForm(p => ({...p, active: !p.active}))}
                    className={`w-12 h-6 rounded-full transition-all relative ${newBannerForm.active ? 'bg-green-500' : 'bg-slate-200'}`}
                  >
                    <div className={`absolute top-0.5 w-5 h-5 bg-white rounded-full shadow transition-all ${newBannerForm.active ? 'left-6' : 'left-0.5'}`}></div>
                  </button>
                </div>

                <button onClick={saveBanner} className="w-full py-4 bg-slate-900 text-white rounded-2xl font-black uppercase text-xs tracking-widest hover:bg-black transition-all shadow-xl flex items-center justify-center gap-2">
                  <Save size={16}/> Zapisz baner
                </button>
              </div>
            </div>
          </div>
        )}
      </div>

      {/* ========== DOLNA NAWIGACJA ========== */}
      <nav className="no-print fixed bottom-0 left-0 right-0 bg-white border-t z-40 flex items-center justify-around px-2 py-1.5 shadow-lg">
        {[
          { id: 'home', label: 'Home', icon: Home },
          { id: 'recipes', label: 'Receptury', icon: BookOpen },
          { id: 'my', label: 'Moje', icon: Heart },
          { id: 'account', label: 'Konto', icon: User },
        ].map(tab => {
          const Icon = tab.icon;
          const isActive = activeTab === tab.id && !showCalculator;
          return (
            <button
              key={tab.id}
              onClick={() => { setActiveTab(tab.id); setShowCalculator(false); }}
              className={`flex flex-col items-center gap-0.5 px-5 py-2 rounded-2xl transition-all ${isActive ? 'text-blue-600' : 'text-slate-400 hover:text-slate-600'}`}
            >
              <Icon size={22} strokeWidth={isActive ? 2.5 : 1.5} fill={isActive && tab.id === 'my' ? 'currentColor' : 'none'} />
              <span className={`text-[9px] font-black uppercase tracking-wider ${isActive ? 'text-blue-600' : 'text-slate-400'}`}>{tab.label}</span>
            </button>
          );
        })}
      </nav>

      {/* ========== MODAL AUTH ========== */}
      {isAuthModalOpen && (
        <div className="no-print fixed inset-0 bg-[#0F172A]/95 z-50 flex items-center justify-center p-4 backdrop-blur-xl animate-in fade-in duration-300">
          <div className="bg-white rounded-[3rem] p-10 md:p-12 max-w-md w-full relative shadow-2xl border-t-[16px] border-blue-600">
            <button onClick={() => setIsAuthModalOpen(false)} className="absolute top-7 right-7 text-slate-300 hover:text-slate-900 transition-colors"><X size={28}/></button>
            <div className="w-16 h-12 mx-auto mb-6 drop-shadow-xl"><img src="/logo.svg" alt="Logo" className="w-full h-full" /></div>
            <h2 className="text-3xl font-black uppercase tracking-tighter mb-6 italic text-slate-900 leading-none text-center">Witaj w EBRA</h2>
            <div className="space-y-3">
              <button onClick={handleGoogleLogin} className="w-full flex items-center justify-center gap-4 border-2 p-4 rounded-2xl font-black uppercase text-[10px] hover:bg-slate-50 transition-all shadow-sm"><img src="https://www.gstatic.com/firebasejs/ui/2.0.0/images/auth/google.svg" className="w-4 h-4" alt="G" /> Google</button>
              <button onClick={handleFacebookLogin} className="w-full flex items-center justify-center gap-4 bg-[#1877F2] text-white p-4 rounded-2xl font-black uppercase text-[10px] shadow-lg hover:brightness-110 transition-all"><img src="https://www.gstatic.com/firebasejs/ui/2.0.0/images/auth/facebook.svg" className="w-4 h-4" alt="F" /> Facebook</button>
            </div>
            <div className="relative my-8 border-t border-slate-100 flex justify-center text-center text-slate-300 uppercase font-black text-[10px]"><span className="bg-white px-4 -mt-2.5 tracking-widest leading-none">Lub E-mail</span></div>
            <form onSubmit={handleEmailAuth} className="space-y-3 text-left text-slate-800">
              <input type="email" required placeholder="E-mail" className="w-full border-2 p-4 rounded-2xl font-bold bg-slate-50 outline-none focus:border-blue-500" value={email} onChange={e=>setEmail(e.target.value)} />
              <div className="flex flex-col gap-2">
                <input type="password" required placeholder="Hasło" className="w-full border-2 p-4 rounded-2xl font-bold bg-slate-50 outline-none focus:border-blue-500" value={password} onChange={e=>setPassword(e.target.value)} />
                <button type="button" onClick={handleResetPassword} className="text-left text-[9px] font-bold text-blue-500 uppercase px-4 leading-none">Zapomniałeś hasła?</button>
              </div>
              <button type="submit" className="w-full bg-[#0F172A] text-white py-5 rounded-2xl font-black uppercase text-xs tracking-widest shadow-2xl active:scale-95 transition-all mt-4">{authMode === 'login' ? 'Zaloguj się' : 'Załóż konto'}</button>
            </form>
            <button onClick={() => setAuthMode(authMode === 'login' ? 'register' : 'login')} className="mt-6 text-[10px] font-bold text-blue-500 uppercase underline tracking-widest text-center w-full block">{authMode === 'login' ? 'Nie masz konta? Zarejestruj się' : 'Masz już konto? Zaloguj się'}</button>
          </div>
        </div>
      )}

      {/* ========== MODAL DODAWANIA / EDYCJI ========== */}
      {isAddModalOpen && (
        <div className="no-print fixed inset-0 bg-slate-900/95 z-50 flex items-center justify-center p-4 backdrop-blur-xl animate-in zoom-in-95 duration-200">
          <div className="bg-white rounded-[4rem] w-full max-w-7xl max-h-[90vh] overflow-y-auto p-8 md:p-10 relative">
            <button onClick={()=>setIsAddModalOpen(false)} className="absolute top-7 right-7 p-2 bg-slate-100 rounded-full hover:bg-slate-200 transition-colors"><X size={20}/></button>
            <h2 className="text-3xl font-black uppercase mb-8 italic text-left leading-none tracking-tighter">Zarządzanie Recepturą</h2>
            <div className="grid grid-cols-1 lg:grid-cols-2 gap-8 text-left text-slate-800">
              <div className="space-y-4">
                <div className="space-y-1"><p className="text-[10px] font-black text-slate-300 uppercase ml-4">Nazwa wyrobu</p><input placeholder="Np. Krakowska Sucha" className="w-full p-5 border-2 rounded-2xl font-bold focus:border-blue-500 outline-none" value={formRecipe.name} onChange={e=>setFormRecipe({...formRecipe, name: e.target.value})} /></div>
                <div className="space-y-1"><p className="text-[10px] font-black text-slate-300 uppercase ml-4">Kategoria</p><select className="w-full p-5 border-2 rounded-2xl font-bold cursor-pointer outline-none" value={formRecipe.category} onChange={e=>setFormRecipe({...formRecipe, category: e.target.value})}><option value="">Wybierz...</option>{categories.map(c=><option key={c} value={c}>{c}</option>)}</select></div>
                <div className="space-y-1"><p className="text-[10px] font-black text-slate-300 uppercase ml-4 leading-none">Proces (użyj **pogrubienie**)</p><textarea placeholder="Opis procesu produkcyjnego..." className="w-full p-5 border-2 rounded-2xl h-56 font-medium outline-none focus:border-blue-500 text-left" value={formRecipe.tech} onChange={e=>setFormRecipe({...formRecipe, tech: e.target.value})} /></div>
                <div className="bg-slate-50 p-6 rounded-[2.5rem] border-2 border-dashed text-center min-h-[120px] flex items-center justify-center relative overflow-hidden text-slate-400">{formRecipe.imageUrl ? (<><img src={formRecipe.imageUrl} className="w-full h-28 object-cover rounded-2xl shadow-lg" alt="P"/><button onClick={() => setFormRecipe({...formRecipe, imageUrl: ''})} className="absolute top-2 right-2 bg-red-600 text-white p-1.5 rounded-full shadow-lg"><X size={12}/></button></>) : (<label className="cursor-pointer w-full"><svg xmlns="http://www.w3.org/2000/svg" width="36" height="36" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" className="mx-auto mb-2 opacity-30"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"/><polyline points="17 8 12 3 7 8"/><line x1="12" y1="3" x2="12" y2="15"/></svg><span className="text-[9px] font-black uppercase tracking-widest">{isUploading ? 'Wgrywanie...' : 'Wgraj zdjęcie'}</span><input type="file" className="hidden" accept="image/*" onChange={handleMyDevilUpload} disabled={isUploading}/></label>)}</div>
              </div>

              <div className="space-y-5">
                <div className="flex justify-between items-center font-black uppercase text-slate-400 ml-2 tracking-widest">Surowce (KG)<button onClick={()=>setFormRecipe({...formRecipe, meats: [...formRecipe.meats, {name: '', val: 0, grinding: ''}]})} className="p-2 bg-blue-600 text-white rounded-lg shadow-lg"><Plus size={15}/></button></div>
                {formRecipe.meats.map((m, i) => (
                  <div key={i} className="flex gap-2">
                    <input placeholder="Mięso" className="flex-1 p-3 border-2 rounded-xl font-bold text-sm" value={m.name} onChange={e=>{const l=[...formRecipe.meats]; l[i].name=e.target.value; setFormRecipe({...formRecipe, meats: l})}} />
                    <input placeholder="Sitko" className="w-20 p-3 border-2 rounded-xl text-center text-xs" value={m.grinding} onChange={e=>{const l=[...formRecipe.meats]; l[i].grinding=e.target.value; setFormRecipe({...formRecipe, meats: l})}} />
                    <div className="relative"><input type="number" className="w-20 p-3 border-2 rounded-xl text-center font-black text-blue-600 pr-5" value={m.val ?? m.percentage ?? ''} onChange={e=>{const l=[...formRecipe.meats]; l[i].val=e.target.value; setFormRecipe({...formRecipe, meats: l})}} /><span className="absolute right-2 top-3.5 text-[8px] font-black text-slate-300">KG</span></div>
                    <button onClick={()=>{const l=[...formRecipe.meats]; l.splice(i,1); setFormRecipe({...formRecipe, meats: l})}} className="text-red-300 hover:text-red-600 p-1"><Trash2 size={15}/></button>
                  </div>
                ))}
                <div className="bg-slate-50 p-3 rounded-2xl text-[10px] font-black uppercase text-slate-400 flex justify-between"><span>Suma wag:</span><span className="text-blue-600">{formRecipe.meats.reduce((acc, m) => acc + Number(m.val ?? m.percentage ?? 0), 0).toFixed(2)} KG</span></div>

                <div className="flex justify-between items-center font-black uppercase text-slate-400 ml-2 tracking-widest mt-4">Przyprawy (g/kg)<button onClick={()=>setFormRecipe({...formRecipe, spices: [...formRecipe.spices, {name: '', perKg: 0, unit: 'g'}]})} className="p-2 bg-slate-900 text-white rounded-lg shadow-lg"><Plus size={15}/></button></div>
                {formRecipe.spices.map((s, i) => (
                  <div key={i} className="flex gap-2">
                    <input placeholder="Nazwa" className="flex-1 p-3 border-2 rounded-xl font-bold text-sm" value={s.name} onChange={e=>{const l=[...formRecipe.spices]; l[i].name=e.target.value; setFormRecipe({...formRecipe, spices: l})}} />
                    <input type="number" step="0.1" className="w-20 p-3 border-2 rounded-xl text-center font-black" value={s.perKg} onChange={e=>{const l=[...formRecipe.spices]; l[i].perKg=e.target.value; setFormRecipe({...formRecipe, spices: l})}} />
                    <select className="w-16 p-3 border-2 rounded-xl text-center font-bold text-xs bg-white outline-none" value={s.unit} onChange={e=>{const l=[...formRecipe.spices]; l[i].unit=e.target.value; setFormRecipe({...formRecipe, spices: l})}}>
                      <option value="g">g</option>
                      <option value="łyż.">łyż.</option>
                      <option value="ml">ml</option>
                      <option value="l">l</option>
                      <option value="szt">szt</option>
                    </select>
                    <button onClick={()=>{const l=[...formRecipe.spices]; l.splice(i,1); setFormRecipe({...formRecipe, spices: l})}} className="text-red-300 hover:text-red-600 p-1"><Trash2 size={15}/></button>
                  </div>
                ))}
              </div>
            </div>
            <button onClick={handleSaveRecipe} className="w-full mt-8 py-5 bg-slate-900 text-white rounded-3xl font-black uppercase tracking-widest shadow-2xl hover:bg-black active:scale-95 transition-all flex items-center justify-center gap-2"><Save size={18}/> Zapisz Recepturę</button>
          </div>
        </div>
      )}
    </div>
  );
};

export default App;
